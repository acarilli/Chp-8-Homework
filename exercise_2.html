peace <- read_csv("PeaceCorpsHW.csv") %>% 
  clean_names() %>% 
  rename(unem_rate = unemployrate,

  mutate(state_short = state2abbr(state),
         year = factor(year),
         state = factor(state)) %>% 
  select(state, year, state_short, apps_pc, unem_rate)
